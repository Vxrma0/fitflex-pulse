```tsx
import React, { useEffect, useState } from "react";
import { useBle } from "../hooks/useBle";
import { RepDetector } from "../lib/repDetector";
import { uploadSession } from "../services/sessionService";
import "./LiveTrackingPanel.css";

export const LiveTrackingPanel: React.FC = () => {
  const { deviceName, batteryPercent, lastPacket, connectionState, startScan, disconnect } = useBle();
  const [detector] = useState(() => new RepDetector());
  const [repCount, setRepCount] = useState(0);
  const [setCount, setSetCount] = useState(0);
  const [exercise, setExercise] = useState("Bicep Curl");
  const [weightKg, setWeightKg] = useState<number>(12.5);
  const [inSession, setInSession] = useState(false);
  const [sessionStart, setSessionStart] = useState<number | null>(null);

  useEffect(() => {
    const sub = detector.onChange((s) => {
      setRepCount(s.repCount);
      setSetCount(s.setCount);
    });
    return () => sub();
  }, [detector]);

  // feed detector with incoming telemetry (simulator or real)
  useEffect(() => {
    if (!lastPacket) return;
    try {
      const ts = lastPacket.timestamp ?? Date.now();
      const ax = lastPacket.accel?.x ?? 0;
      const ay = lastPacket.accel?.y ?? 0;
      const az = lastPacket.accel?.z ?? 0;
      detector.pushSample(ts, ax, ay, az);
    } catch (e) {
      // ignore malformed packets
    }
  }, [lastPacket, detector]);

  const startWorkout = () => {
    detector.reset();
    setRepCount(0);
    setSetCount(0);
    setInSession(true);
    setSessionStart(Date.now());
  };

  const endWorkout = async () => {
    if (!inSession || sessionStart == null) return;
    const summary = {
      startTime: sessionStart,
      endTime: Date.now(),
      exercises: [
        {
          exerciseId: exercise.toLowerCase().replace(/\s+/g, "_"),
          name: exercise,
          sets: [
            {
              setNumber: 1,
              reps: repCount,
              weightKg,
              durationMs: Date.now() - sessionStart,
            },
          ],
          totalVolumeKg: repCount * weightKg,
        },
      ],
      summaryMetrics: {
        totalReps: repCount,
        totalSets: setCount || 1,
        totalVolumeKg: repCount * weightKg,
      },
      device: {
        deviceName,
        batteryPercent,
      },
      uploadedAt: Date.now(),
    };

    await uploadSession(summary);
    setInSession(false);
    setSessionStart(null);
  };

  return (
    <div className="lf-panel">
      <div className="lf-header">
        <div className="lf-title">
          <img src="/fitflex-logo.png" alt="FitFlex" className="lf-logo" />
          <div>
            <div className="lf-exercise">{exercise}</div>
            <div className="lf-sub">Single-arm isolation • Dumbbell</div>
          </div>
        </div>
        <div className="lf-conn">
          <div className={`lf-dot ${connectionState === "connected" ? "on" : "off"}`} />
          <div className="lf-conn-text">{connectionState}</div>
        </div>
      </div>

      <div className="lf-body">
        <div className="lf-ring">
          <svg viewBox="0 0 120 120" className="lf-svg">
            <circle cx="60" cy="60" r="48" className="lf-bg" />
            <circle
              cx="60"
              cy="60"
              r="48"
              className="lf-fg"
              style={{ strokeDasharray: 302, strokeDashoffset: 302 - (repCount % 10) * 30.2 }}
            />
          </svg>
          <div className="lf-counter">
            <div className="lf-count">{repCount}</div>
            <div className="lf-set">Set {setCount + 1}</div>
          </div>
        </div>
      </div>

      <div className="lf-controls">
        <div className="lf-weight">
          <label>Weight</label>
          <input
            type="number"
            value={weightKg}
            onChange={(e) => setWeightKg(parseFloat(e.target.value || "0"))}
            step="0.5"
            min="0"
          />
          <span>kg</span>
        </div>

        <div className="lf-actions">
          <button onClick={startScan}>Scan Band</button>
          <button onClick={() => (connectionState === "connected" ? disconnect() : startScan())}>
            {connectionState === "connected" ? "Disconnect" : "Connect"}
          </button>
          {!inSession ? (
            <button className="primary" onClick={startWorkout}>
              Start Workout
            </button>
          ) : (
            <button className="danger" onClick={endWorkout}>
              End Workout
            </button>
          )}
        </div>
      </div>

      <div className="lf-footer">
        <div>{deviceName || "No band paired"}</div>
        <div>Battery: {batteryPercent}%</div>
      </div>
    </div>
  );
};

export default LiveTrackingPanel;
```

#### `src/hooks/useBle.tsx`
```tsx
import { useEffect, useState } from "react";

/**
 * Web demo BLE hook (simulator-friendly).
 * In production mobile app, BLE is handled natively (Flutter).
 *
 * This hook exposes:
 * - deviceName
 * - batteryPercent
 * - lastPacket (JSON telemetry)
 * - connectionState: 'disconnected'|'scanning'|'connected'
 * - startScan(), disconnect()
 *
 * For the web demo we support a simulator that writes to window.__BLE_SIMULATOR_PUSH(packet)
 */

type Telemetry = {
  timestamp?: number;
  accel?: { x: number; y: number; z: number };
  gyro?: { x: number; y: number; z: number };
  battery?: number;
};

export function useBle() {
  const [deviceName, setDeviceName] = useState<string>("");
  const [batteryPercent, setBatteryPercent] = useState<number>(0);
  const [lastPacket, setLastPacket] = useState<Telemetry | null>(null);
  const [connectionState, setConnectionState] = useState<"disconnected" | "scanning" | "connected">("disconnected");

  useEffect(() => {
    // simulator hook: window.__BLE_SIMULATOR_PUSH = (packet) => { ... }
    // eslint-disable-next-line @typescript-eslint/ban-ts-comment
    // @ts-ignore
    window.__BLE_SIMULATOR_PUSH = (packet: Telemetry) => {
      setLastPacket(packet);
      if (packet.battery != null) setBatteryPercent(packet.battery);
      if (!deviceName) setDeviceName("SimBand-001");
      if (connectionState !== "connected") setConnectionState("connected");
    };

    return () => {
      // cleanup
      // eslint-disable-next-line @typescript-eslint/ban-ts-comment
      // @ts-ignore
      delete window.__BLE_SIMULATOR_PUSH;
    };
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, []);

  const startScan = async () => {
    // For web demo, scanning simply flips state and expects simulator to push packets.
    setConnectionState("scanning");
    setTimeout(() => setConnectionState("connected"), 800);
  };

  const disconnect = async () => {
    setConnectionState("disconnected");
    setDeviceName("");
    setBatteryPercent(0);
    setLastPacket(null);
  };

  return { deviceName, batteryPercent, lastPacket, connectionState, startScan, disconnect };
}
```

#### `src/lib/repDetector.ts`
```ts
// Simple rep detection state machine for web demo.
// Uses accelerometer magnitude and derivative heuristics.
// Exposes onChange(callback) to subscribe to state updates.

type Sample = { ts: number; mag: number };

export class RepDetector {
  private buffer: Sample[] = [];
  private repCount = 0;
  private setCount = 0;
  private state: "idle" | "up" | "top" | "down" | "complete" = "idle";
  private lastRepTs = 0;
  private listeners: Array<(s: { repCount: number; setCount: number }) => void> = [];

  // tunables
  amplitudeThreshold = 0.12;
  minRepDurationMs = 300;
  maxRepDurationMs = 8000;
  setTimeoutSeconds = 10;

  private setTimer?: number;

  pushSample(ts: number, ax: number, ay: number, az: number) {
    const mag = Math.sqrt(ax * ax + ay * ay + az * az);
    this.buffer.push({ ts, mag });
    if (this.buffer.length > 500) this.buffer.splice(0, this.buffer.length - 500);
    this.process();
  }

  reset() {
    this.buffer = [];
    this.repCount = 0;
    this.setCount = 0;
    this.state = "idle";
    this.lastRepTs = 0;
    if (this.setTimer) {
      clearTimeout(this.setTimer);
      this.setTimer = undefined;
    }
    this.emit();
  }

  onChange(cb: (s: { repCount: number; setCount: number }) => void) {
    this.listeners.push(cb);
    cb({ repCount: this.repCount, setCount: this.setCount });
    return () => {
      this.listeners = this.listeners.filter((l) => l !== cb);
    };
  }

  private emit() {
    const snapshot = { repCount: this.repCount, setCount: this.setCount };
    this.listeners.forEach((l) => l(snapshot));
  }

  private process() {
    if (this.buffer.length < 3) return;
    const last = this.buffer[this.buffer.length - 1];
    const prev = this.buffer[this.buffer.length - 2];
    const prev2 = this.buffer[this.buffer.length - 3];
    const derivative = last.mag - prev.mag;

    switch (this.state) {
      case "idle":
        if (derivative > this.amplitudeThreshold / 4) this.state = "up";
        break;
      case "up":
        if (derivative < 0 && last.mag > prev.mag && last.mag > this.amplitudeThreshold) this.state = "top";
        break;
      case "top":
        if (derivative < -this.amplitudeThreshold / 8) this.state = "down";
        break;
      case "down":
        if (Math.abs(derivative) < 0.01) {
          const now = last.ts;
          const duration = this.lastRepTs === 0 ? now : now - this.lastRepTs;
          if (this.lastRepTs === 0 || (duration >= this.minRepDurationMs && duration <= this.maxRepDurationMs)) {
            this.registerRep(now);
          }
          this.state = "complete";
        }
        break;
      case "complete":
        if (Math.abs(derivative) < 0.005) this.state = "idle";
        break;
    }
  }

  private registerRep(ts: number) {
    this.repCount += 1;
    this.lastRepTs = ts;
    if (this.setTimer) {
      clearTimeout(this.setTimer);
    }
    this.setTimer = window.setTimeout(() => {
      if (this.repCount > 0) {
        this.setCount += 1;
        this.repCount = 0;
        this.emit();
      }
    }, this.setTimeoutSeconds * 1000);
    this.emit();
  }
}
```

```tsx
import React, { useRef } from "react";

/**
 * Simple simulator UI for feeding CSV/JSON telemetry into the web demo.
 * Usage: paste CSV content into textarea and press Play.
 *
 * It will call window.__BLE_SIMULATOR_PUSH(packet) for each row.
 */

export const BleSimulator: React.FC = () => {
  const textareaRef = useRef<HTMLTextAreaElement | null>(null);
  const playingRef = useRef<number | null>(null);

  const playCsv = async () => {
    const text = textareaRef.current?.value ?? "";
    if (!text) return;
    const lines = text.split(/\r?\n/).map((l) => l.trim()).filter(Boolean);
    // detect header
    let start = 0;
    if (lines[0].toLowerCase().includes("timestamp")) start = 1;
    const rows = lines.slice(start).map((l) => l.split(",").map((c) => c.trim()));
    let i = 0;
    const push = () => {
      if (i >= rows.length) {
        if (playingRef.current) {
          clearInterval(playingRef.current);
          playingRef.current = null;
        }
        return;
      }
      const r = rows[i];
      const ts = parseInt(r[0], 10) || Date.now();
      const ax = parseFloat(r[1]) || 0;
      const ay = parseFloat(r[2]) || 0;
      const az = parseFloat(r[3]) || 0;
      const gx = parseFloat(r[4]) || 0;
      const gy = parseFloat(r[5]) || 0;
      const gz = parseFloat(r[6]) || 0;
      // eslint-disable-next-line @typescript-eslint/ban-ts-comment
      // @ts-ignore
      if (typeof window.__BLE_SIMULATOR_PUSH === "function") {
        // push JSON telemetry
        // battery simulated as 85
        window.__BLE_SIMULATOR_PUSH({
          timestamp: ts,
          accel: { x: ax, y: ay, z: az },
          gyro: { x: gx, y: gy, z: gz },
          battery: 85,
        });
      }
      i++;
    };
    // play at 100Hz or based on timestamps (simple fixed rate here)
    playingRef.current = window.setInterval(push, 10);
  };

  const stop = () => {
    if (playingRef.current) {
      clearInterval(playingRef.current);
      playingRef.current = null;
    }
  };

  return (
    <div style={{ marginTop: 12 }}>
      <div style={{ marginBottom: 8, color: "#9aa6ad" }}>BLE Simulator (CSV)</div>
      <textarea ref={textareaRef} style={{ width: "100%", height: 120, background: "#071018", color: "#e6eef0", padding: 8 }} defaultValue={`timestamp,ax,ay,az,gx,gy,gz
1717563890000,0.01,-0.98,0.02,0.01,0.02,0.00
1717563890100,0.05,-0.90,0.10,0.02,0.03,0.00
1717563890200,0.20,-0.60,0.30,0.03,0.05,0.01
1717563890300,0.40,-0.20,0.60,0.05,0.10,0.02
1717563890400,0.10,-0.80,0.20,0.02,0.03,0.00
1717563890500,0.02,-0.98,0.01,0.01,0.01,0.00`} />
      <div style={{ display: "flex", gap: 8, marginTop: 8 }}>
        <button onClick={playCsv}>Play</button>
        <button onClick={stop}>Stop</button>
      </div>
    </div>
  );
};

export default BleSimulator;
```

#### `src/services/sessionService.ts`
```ts
import { getAuth } from "firebase/auth";
import { getFirestore, collection, doc, setDoc } from "firebase/firestore";

/**
 * uploadSession(summary)
 * summary: object matching the schema described in the project plan.
 *
 * This function assumes Firebase has been initialized in your app.
 */

export async function uploadSession(summary: any) {
  try {
    const auth = getAuth();
    const user = auth.currentUser;
    if (!user) {
      // For demo: store under 'anon' user
      const db = getFirestore();
      const docRef = doc(collection(db, "users", "anon", "sessions"));
      await setDoc(docRef, summary);
      return { ok: true, id: docRef.id };
    } else {
      const db = getFirestore();
      const docRef = doc(collection(db, "users", user.uid, "sessions"));
      await setDoc(docRef, summary);
      return { ok: true, id: docRef.id };
    }
  } catch (e) {
    console.error("uploadSession error", e);
    return { ok: false, error: e };
  }
}
```
